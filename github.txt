1. git init - команда, которая инициализирует пустой локальный репозиторий(создает скрытую папку, в которой все будет хранить)
2. git remote -v - посмотреть привязки к удаленным репозиториям
3. git remote add origin 'SSH or HTTPS' - создание привязки к определенному репозиторию
4. git remote remove origin - удаление привязки к репозиторию
5. git status - посмотреть статус файлов(отслеживаются или нет)
6. git commit -m 'Some message' - маркировка коммита(описание должно быть кратким и содержательным)
7. git push origin branch_name - отправка данных на определенному ветку
8. git pull origin branch_name - стянуть данных с определенной ветки
9. git stash - спрятать локальные изменение(обычно применяется перед стягиванием изменений с другой ветки)
10. git stash apply - вернуть спрятанные ранее изменение(добавить свои поверх изменение, которые стянули с другой ветки)
11. git branch -v - посмотреть все ветки и на какой ветке находимся щяс
12. git branch branch_name - создание новой ветки
13. git checkout branch_name - переключение на другую ветку
14. git branch --delete branch_name - удаление ветки
